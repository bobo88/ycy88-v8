    // 方法二：优化方法一
    let aLen = a.length
    let bLen = b.length
    let res: string[] = []
    let isEnter: number = 0
    while (aLen > 0 || bLen > 0 || isEnter) {
      let numA = +(a.charAt(aLen - 1) || '0')
      let numB = +(b.charAt(bLen - 1) || '0')
      let addRes = numA + numB + isEnter // 结果只有 0 1 2 3 四种情况
      if (addRes > 1) {
        res.unshift(addRes % 2 === 1 ? '1' : '0')
        isEnter = 1
      } else if (addRes === 1) {
        res.unshift('1')
        isEnter = 0
      } else {
        // 全部等于0
        res.unshift('0')
        isEnter = 0
      }
      aLen--
      bLen--
    }
    return res.join('')
  }
